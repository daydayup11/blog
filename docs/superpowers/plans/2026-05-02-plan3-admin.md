# CMS Admin Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the CMS admin backend — login, stats dashboard, post/column/project/garden management, and Markdown editor — all as static HTML pages that call the existing Go API.

**Architecture:** Pure static HTML/CSS/JS under `frontend/admin/`, same pattern as public pages. All pages check for a valid JWT in `localStorage` on load and redirect to login if missing. A shared `admin/js/auth.js` handles the JWT guard and logout. Pages call `api.js` methods (already include the `Authorization` header automatically). No new backend code needed.

**Tech Stack:** HTML5, CSS3, Vanilla JS ES6 modules, existing `css/common.css` + `css/theme-*.css`, `js/api.js` (already has all admin methods), EasyMDE (CDN) for Markdown editor.

---

## File Structure

```
frontend/admin/
├── index.html          # Login page — username/password → JWT → redirect to dashboard
├── dashboard.html      # Stats: PV/UV chart + top posts + GA4 link

├── garden.html         # Garden sections + items management
└── js/
    └── auth.js         # JWT guard (redirect if no token) + logout helper
```

All pages share:
- `../../css/theme-dark.css`, `../../css/theme-light.css`, `../../css/common.css`
- `../../js/theme.js` for dark/light toggle
- `./js/auth.js` for JWT guard
- `../../js/api.js` for all API calls (imported as ES module)

Create `frontend/admin/js/auth.js`:

```js
export function requireAuth() {
  if (!localStorage.getItem('admin-token')) {
    location.href = '/admin/index.html';
  }
}

export function logout() {
  localStorage.removeItem('admin-token');
  location.href = '/admin/index.html';
}

export function renderAdminNav(activePage) {
  const pages = [
    { href: '/admin/dashboard.html', label: '统计',   key: 'dashboard' },
    { href: '/admin/posts.html',     label: '文章',   key: 'posts'     },
    { href: '/admin/columns.html',   label: '专栏',   key: 'columns'   },
    { href: '/admin/projects.html',  label: '项目',   key: 'projects'  },
    { href: '/admin/garden.html',    label: '花园',   key: 'garden'    },
  ];
  const links = pages.map(p =>
    `<a href="${p.href}" class="${p.key === activePage ? 'active' : ''}">${p.label}</a>`
  ).join('');
  return `
    <nav class="nav">
      <span class="nav-logo" style="font-size:14px">dyt <span style="color:var(--text-muted);font-weight:400">/ 后台</span></span>
      <div class="nav-links">${links}</div>
      <a href="/" style="font-size:13px;color:var(--text-muted);margin-right:8px">← 前台</a>
      <button class="nav-theme-btn" id="theme-btn">🌙</button>
      <button onclick="import('/admin/js/auth.js').then(m=>m.logout())"
        style="background:none;border:1px solid var(--border);border-radius:6px;padding:4px 10px;cursor:pointer;font-size:13px;color:var(--text-muted);margin-left:8px">
        退出
      </button>
    </nav>`;
}
```

```html
    import { requireAuth, renderAdminNav } from '/admin/js/auth.js';
    import { api } from '/js/api.js';
    import '/js/theme.js';

    document.getElementById('nav-root').innerHTML = renderAdminNav('dashboard');

    let currentDays = 7;

    async function loadStats(days) {
      currentDays = days;
      document.querySelectorAll('[data-days]').forEach(b =>
        b.classList.toggle('active', +b.dataset.days === days));

    function drawChart(daily) {
      const canvas = document.getElementById('stats-chart');
      const ctx = canvas.getContext('2d');
      canvas.width = canvas.parentElement.clientWidth - 40;
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      if (!daily.length) {
        ctx.fillStyle = 'var(--text-muted)';
        ctx.font = '13px system-ui';
        ctx.fillText('暂无数据', canvas.width / 2 - 24, canvas.height / 2);
        return;
      }

      const maxVal = Math.max(...daily.map(d => Math.max(d.pv, d.uv)), 1);
      const barW = Math.floor((canvas.width - 20) / daily.length / 2.5);
      const gap  = Math.floor((canvas.width - 20) / daily.length);
      const h    = canvas.height - 24;

      const isDark = document.documentElement.getAttribute('data-theme') === 'dark';
      const pvColor = isDark ? '#00ff88' : '#2563eb';
      const uvColor = isDark ? '#60a5fa' : '#93c5fd';

      daily.forEach((d, i) => {
        const x = 10 + i * gap;
        const pvH = Math.round((d.pv / maxVal) * h);
        const uvH = Math.round((d.uv / maxVal) * h);

        ctx.fillStyle = pvColor;
        ctx.fillRect(x, h - pvH, barW, pvH);
        ctx.fillStyle = uvColor;
        ctx.fillRect(x + barW + 2, h - uvH, barW, uvH);

        if (i % Math.ceil(daily.length / 7) === 0) {
          ctx.fillStyle = isDark ? '#475569' : '#94a3b8';
          ctx.font = '10px system-ui';
          ctx.fillText(d.date.slice(5), x, canvas.height);
        }
      });
    }

    document.querySelector('[data-days]').parentElement.addEventListener('click', e => {
      if (e.target.matches('[data-days]')) loadStats(+e.target.dataset.days);
    });

    loadStats(7);
  </script>

</html>
```

```html
      // Fetch all pages
      let allPosts = [], page = 1;

        return;

              ${p.is_published ? '已发布' : '草稿'}

            <button class="action-btn ${p.is_published ? 'active-btn' : ''}" onclick="togglePublish(${p.id}, ${p.is_published})">

            </button>
            <button class="action-btn danger" onclick="deletePost(${p.id})">删除</button>
          </td>
        </tr>`).join('');

    window.editPost = (id, slug) => {
      location.href = `/admin/editor.html?slug=${slug}`;

    };

  </script>
```

## Task 8: MD 文件导入 + PDF 上传支持

**涉及文件：**
- Modify: `frontend/admin/editor.html` — 加"导入 MD 文件"按钮
- Modify: `frontend/admin/garden.html` — 条目表单加 PDF 上传

- Modify: `frontend/post.html` — 正文中 PDF 链接自动转 embed（可选）

**原理：**
- MD 导入：纯前端 `FileReader` 读取 `.md` 文件内容 → 填入编辑框，不改后端
- PDF 上传：复用现有 `POST /api/admin/upload`（后端已支持任意文件），返回 `/uploads/xxx.pdf`，存入 `embed_url` 字段
- PDF 展示：前端检查 `embed_url` 是否以 `.pdf` 结尾，是则用 `<embed>` 替代 `<iframe>`
